 * version number を取得します
 * @returns {string} version number を返します

  Touching,
};

MOKU.net = {
  Ajax,
  Cookie,
  Queries,
};
/**
 * MOKU.tick
 * @type {Object} MOKU.tick object を返します
 */
MOKU.tick = {
  Cycle,
  Fps,
  Polling,
  Rate,
};
/**
 * MOKU.util
 * @type {Object} MOKU.util object を返します

  Type,
  List,
  Text,

  Iro,

  Bounding,
  Classes,
  Elements,
};

/**
 * global object `MOKU`
 * @type {Object}

window.MOKU = MOKU;
